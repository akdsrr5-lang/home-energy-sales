const SHEET_ID = '1Zq1Xx8VRxqIybDUfbUmMO6YMKGebOI2JGO2GA7f3qDU'; function doGet(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.openById(SHEET_ID); if (action === 'load') { return output({ properties: getSheet(ss, 'properties'), budgets: getSheet(ss, 'budgets'), gas: getSheet(ss, 'gas'), locks: getSheet(ss, 'locks'), }); } if (action === 'save_each') { const key = e.parameter.key; const raw = e.parameter.val; if (!key || !raw) return output({ error: 'missing key or val' }); const val = JSON.parse(decodeURIComponent(raw)); saveSheet(ss, key, val); return output({ ok: true, key: key }); } return output({ error: 'unknown action: ' + action }); } catch(err) { return output({ error: err.toString() }); } } function doPost(e) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const body = JSON.parse(e.postData.contents); if (body.action === 'load') { return output({ properties: getSheet(ss, 'properties'), budgets: getSheet(ss, 'budgets'), gas: getSheet(ss, 'gas'), locks: getSheet(ss, 'locks'), }); } if (body.action === 'save_each') { saveSheet(ss, body.key, body.val); return output({ ok: true }); } } catch(err) { return output({ error: err.toString() }); } return output({ error: 'unknown' }); } function getSheet(ss, name) { const sheet = ss.getSheetByName(name); if (!sheet || sheet.getLastRow() < 1) { return (name === 'budgets' || name === 'locks') ? {} : []; } const val = sheet.getRange(1, 1).getValue(); try { return JSON.parse(val); } catch(e) { return (name === 'budgets' || name === 'locks') ? {} : []; } } function saveSheet(ss, name, data) { let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); sheet.clearContents(); sheet.getRange(1, 1).setValue(JSON.stringify(data)); SpreadsheetApp.flush(); } function output(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
